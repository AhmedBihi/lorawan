107 void
108 LoraChannel::Send (Ptr< LoraPhy > sender, Ptr< Packet > packet,
109  double txPowerDbm, LoraTxParameters txParams,
110  Time duration, double frequencyMHz) const
111 {
112  NS_LOG_FUNCTION (this << sender << packet << txPowerDbm << txParams <<
113  duration << frequencyMHz);
114 
115  // Get the mobility model of the sender
116  Ptr<MobilityModel> senderMobility = sender->GetMobility ()->GetObject<MobilityModel> ();
117 
118  NS_ASSERT (senderMobility != 0); // Make sure it's available
119 
120  NS_LOG_INFO ("Starting cycle over all " << m_phyList.size () << " PHYs");
121  NS_LOG_INFO ("Sender mobility: " << senderMobility->GetPosition ());
122 
123  // Cycle over all registered PHYs
124  uint32_t j = 0;
125  std::vector<Ptr<LoraPhy> >::const_iterator i;
126  for (i = m_phyList.begin (); i != m_phyList.end (); i++, j++)
127  {
128  // Do not deliver to the sender (*i is the current PHY)
129  if (sender != (*i))
130  {
131  // Get the receiver's mobility model
132  Ptr<MobilityModel> receiverMobility = (*i)->GetMobility ()->
133  GetObject<MobilityModel> ();
134 
135  NS_LOG_INFO ("Receiver mobility: " <<
136  receiverMobility->GetPosition ());
137 
138  // Compute delay using the delay model
139  Time delay = m_delay->GetDelay (senderMobility, receiverMobility);
140 
141  // Compute received power using the loss model
142  double rxPowerDbm = GetRxPower (txPowerDbm, senderMobility,
143  receiverMobility);
144 
145  NS_LOG_DEBUG ("Propagation: txPower=" << txPowerDbm <<
146  "dbm, rxPower=" << rxPowerDbm << "dbm, " <<
147  "distance=" << senderMobility->GetDistanceFrom (receiverMobility) <<
148  "m, delay=" << delay);
149 
150  // Get the id of the destination PHY to correctly format the context
151  Ptr<NetDevice> dstNetDevice = m_phyList[j]->GetDevice ();
152  uint32_t dstNode = 0;
153  if (dstNetDevice != 0)
154  {
155  NS_LOG_INFO ("Getting node index from NetDevice, since it exists");
156  dstNode = dstNetDevice->GetNode ()->GetId ();
157  NS_LOG_DEBUG ("dstNode = " << dstNode);
158  }
159  else
160  {
161  NS_LOG_INFO ("No net device connected to the PHY, using context 0");
162  }
163 
164  // Create the parameters object based on the calculations above
165  LoraChannelParameters parameters;
166  parameters.rxPowerDbm = rxPowerDbm;
167  parameters.sf = txParams.sf;
168  parameters.duration = duration;
169  parameters.frequencyMHz = frequencyMHz;
170 
171  // Schedule the receive event
172  NS_LOG_INFO ("Scheduling reception of the packet");
173  Simulator::ScheduleWithContext (dstNode, delay, &LoraChannel::Receive,
174  this, j, packet, parameters);
175 
176  // Fire the trace source for sent packet
177  m_packetSent (packet);
178  }
179  }
180 }
