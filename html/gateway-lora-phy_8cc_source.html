153 void
154 GatewayLoraPhy::Send (Ptr<Packet> packet, LoraTxParameters txParams,
155  double frequencyMHz, double txPowerDbm)
156 {
157  NS_LOG_FUNCTION (this << packet << frequencyMHz << txPowerDbm);
158 
159  // Get the time a packet with these parameters will take to be transmitted
160  Time duration = GetOnAirTime (packet, txParams);
161 
162  /*
163  * Differently from what is done in EndDevices, where packets cannot be
164  * transmitted while in RX state, Gateway sending is assumed to have priority
165  * over reception.
166  *
167  * This different behaviour is motivated by the asymmetry in a typical
168  * LoRaWAN network, where Downlink messages are more critical to network
169  * performance than Uplink ones. Even if the gateway is receiving a packet
170  * on the channel when it is asked to transmit by the upper layer, in order
171  * not to miss the receive window of the end device, the gateway will still
172  * need to send the packet. In order to model this fact, the send event is
173  * registered in the gateway's InterferenceHelper as a received event.
174  * While this may not destroy packets incoming on the same frequency, this
175  * is almost always guaranteed to do so due to the fact that this event can
176  * have a power up to 27 dBm.
177  */
178  m_interference.Add (duration, txPowerDbm, txParams.sf, packet, frequencyMHz);
179 
180  // Send the packet in the channel
181  m_channel->Send (this, packet, txPowerDbm, txParams, duration, frequencyMHz);
182 
183  Simulator::Schedule (duration, &GatewayLoraPhy::TxFinished, this, packet);
184 
185  m_isTransmitting = true;
186 
187  // Fire the trace source
188  if (m_device)
189  {
190  m_startSending (packet, m_device->GetNode ()->GetId ());
191  }
192  else
193  {
194  m_startSending (packet, 0);
195  }
196 }

210 void
211 GatewayLoraPhy::StartReceive (Ptr<Packet> packet, double rxPowerDbm,
212  uint8_t sf, Time duration, double frequencyMHz)
213 {
214  NS_LOG_FUNCTION (this << packet << rxPowerDbm << duration << frequencyMHz);
215 
216  // Fire the trace source
217  m_phyRxBeginTrace (packet);
218 
219  // Add the event to the LoraInterferenceHelper
220  Ptr<LoraInterferenceHelper::Event> event;
221  event = m_interference.Add (duration, rxPowerDbm, sf, packet, frequencyMHz);
222 
223  // Cycle over the receive paths to check availability to receive the packet
224  std::list<Ptr<GatewayLoraPhy::ReceptionPath> >::iterator it;
225 
226  for (it = m_receptionPaths.begin (); it != m_receptionPaths.end (); ++it)
227  {
228  Ptr<GatewayLoraPhy::ReceptionPath> currentPath = *it;
229 
230  NS_LOG_DEBUG ("Current ReceptionPath is centered on frequency = " <<
231  currentPath->GetFrequency ());
232 
233  // If the receive path is available and listening on the channel of
234  // interest, we have a candidate
235  if (currentPath->GetFrequency () == frequencyMHz &&
236  currentPath->IsAvailable ())
237  {
238  // See whether the reception power is above or below the sensitivity
239  // for that spreading factor
240  double sensitivity = GatewayLoraPhy::sensitivity[unsigned(sf)-7];
241 
242  if (rxPowerDbm < sensitivity) // Packet arrived below sensitivity
243  {
244  NS_LOG_INFO ("Dropping packet reception of packet with sf = "
245  << unsigned(sf) <<
246  " because under the sensitivity of "
247  << sensitivity << " dBm");
248 
249  if (m_device)
250  {
251  m_underSensitivity (packet, m_device->GetNode ()->GetId ());
252  }
253  else
254  {
255  m_underSensitivity (packet, 0);
256  }
257 
258  // Since the packet is below sensitivity, it makes no sense to
259  // search for another ReceivePath
260  return;
261  }
262  else // We have sufficient sensitivity to start receiving
263  {
264  NS_LOG_INFO ("Scheduling reception of a packet, " <<
265  "occupying one demodulator");
266 
267  // Block this resource
268  currentPath->LockOnEvent (event);
269  m_occupiedReceptionPaths++;
270 
271  // Schedule the end of the reception of the packet
272  Simulator::Schedule (duration, &LoraPhy::EndReceive, this,
273  packet, event);
274 
275  // Make sure we don't go on searching for other ReceivePaths
276  return;
277  }
278  }
279  }
280  // If we get to this point, there are no demodulators we can use
281  NS_LOG_INFO ("Dropping packet reception of packet with sf = "
282  << unsigned(sf) <<
283  " because no suitable demodulator was found");
284 
285  // Fire the trace source
286  if (m_device)
287  {
288  m_noMoreDemodulators (packet, m_device->GetNode ()->GetId ());
289  }
290  else
291  {
292  m_noMoreDemodulators (packet, 0);
293  }
294 }

296 void
297 GatewayLoraPhy::EndReceive (Ptr<Packet> packet,
298  Ptr<LoraInterferenceHelper::Event> event)
299 {
300  NS_LOG_FUNCTION (this << packet << *event);
301 
302  // Call the trace source
303  m_phyRxEndTrace (packet);
304 
305  // Call the LoraInterferenceHelper to determine whether there was
306  // destructive interference. If the packet is correctly received, this
307  // method returns a 0.
308  uint8_t packetDestroyed = 0;
309  packetDestroyed = m_interference.IsDestroyedByInterference (event);
310 
311  // Check whether the packet was destroyed
312  if (packetDestroyed != uint8_t (0))
313  {
314  NS_LOG_DEBUG ("packetDestroyed by " << unsigned(packetDestroyed));
315 
316  // Update the packet's LoraTag
317  LoraTag tag;
318  packet->RemovePacketTag (tag);
319  tag.SetDestroyedBy (packetDestroyed);
320  packet->AddPacketTag (tag);
321 
322  // Fire the trace source
323  if (m_device)
324  {
325  m_interferedPacket (packet, m_device->GetNode ()->GetId ());
326  }
327  else
328  {
329  m_interferedPacket (packet, 0);
330  }
331  }
332  else // Reception was correct
333  {
334  NS_LOG_INFO ("Packet with SF " <<
335  unsigned(event->GetSpreadingFactor ()) <<
336  " received correctly");
337 
338  // Fire the trace source
339  if (m_device)
340  {
341  m_successfullyReceivedPacket (packet, m_device->GetNode ()->GetId ());
342  }
343  else
344  {
345  m_successfullyReceivedPacket (packet, 0);
346  }
347 
348  // Forward the packet to the upper layer
349  if (!m_rxOkCallback.IsNull ())
350  {
351  // Make a copy of the packet
352  Ptr<Packet> packetCopy = packet->Copy ();
353 
354  // Set the receive power and frequency of this packet in the LoraTag: this
355  // information can be useful for upper layers trying to control link
356  // quality.
357  LoraTag tag;
358  packetCopy->RemovePacketTag (tag);
359  tag.SetReceivePower (event->GetRxPowerdBm ());
360  tag.SetFrequency (event->GetFrequency ());
361  packetCopy->AddPacketTag (tag);
362 
363  m_rxOkCallback (packetCopy);
364  }
365 
366  }
367 
368  // Search for the demodulator that was locked on this event to free it.
369 
370  std::list< Ptr< GatewayLoraPhy::ReceptionPath > >::iterator it;
371 
372  for (it = m_receptionPaths.begin (); it != m_receptionPaths.end (); ++it)
373  {
374  Ptr<GatewayLoraPhy::ReceptionPath> currentPath = *it;
375 
376  if (currentPath->GetEvent () == event)
377  {
378  currentPath->Free ();
379  m_occupiedReceptionPaths--;
380  return;
381  }
382  }
383 }

387 {

389 
390  // Search every demodulator to see whether there's one listening on this
391  // frequency.
392  std::list< Ptr< GatewayLoraPhy::ReceptionPath > >::iterator it;
393 
394  for (it = m_receptionPaths.begin (); it != m_receptionPaths.end (); ++it)
395  {
396  Ptr<GatewayLoraPhy::ReceptionPath> currentPath = *it;
397 
398  NS_LOG_DEBUG ("Current reception path is on frequency " <<

402  {
403  return true;
404  }
405  }
406  return false;
