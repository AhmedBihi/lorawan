96 LoraNetDevice::CompleteConfig (void)
97 {
98  NS_LOG_FUNCTION_NOARGS ();
99 
100  // Verify we have all the necessary pieces
101  if (m_mac == 0 || m_phy == 0 || m_node == 0 || m_configComplete)
102  {
103  return;
104  }
105 
106  m_mac->SetPhy (m_phy);
107  m_configComplete = true;
