91 void
92 EndDeviceLoraPhy::Send (Ptr<Packet> packet, LoraTxParameters txParams,
93  double frequencyMHz, double txPowerDbm)
94 {
95  NS_LOG_FUNCTION (this << packet << txParams << frequencyMHz << txPowerDbm);
96 
97  NS_LOG_INFO ("Current state: " << m_state);
98 
99  // We must be either in STANDBY or SLEEP mode to send a packet
100  if (m_state != STANDBY && m_state != SLEEP)
101  {
102  NS_LOG_INFO ("Cannot send because device is currently not in STANDBY or SLEEP mode");
103  return;
104  }
105 
106  // We can send the packet: switch to the TX state
107  SwitchToTx ();
108 
109  // Compute the duration of the transmission
110  Time duration = GetOnAirTime (packet, txParams);
111 
112  // Tag the packet with information about its Spreading Factor
113  LoraTag tag;
114  packet->RemovePacketTag (tag);
115  tag.SetSpreadingFactor (txParams.sf);
116  packet->AddPacketTag (tag);
117 
118  // Send the packet over the channel
119  NS_LOG_INFO ("Sending the packet in the channel");
120  m_channel->Send (this, packet, txPowerDbm, txParams, duration, frequencyMHz);
121 
122  // Schedule the switch back to STANDBY mode.
123  // For reference see SX1272 datasheet, section 4.1.6
124  Simulator::Schedule (duration, &EndDeviceLoraPhy::SwitchToStandby, this);
125 
126  // Schedule the txFinished callback, if it was set
127  // The call is scheduled just after the switch to standby in case the upper
128  // layer wishes to change the state. This ensures that it will find a PHY in
129  // STANDBY mode.
130  if (!m_txFinishedCallback.IsNull ())
131  {
132  Simulator::Schedule (duration + NanoSeconds (10),
133  &EndDeviceLoraPhy::m_txFinishedCallback, this,
134  packet);
135  }
136 
137 
138  // Call the trace source
139  if (m_device)
140  {
141  m_startSending (packet, m_device->GetNode ()->GetId ());
142  }
143  else
144  {
145  m_startSending (packet, 0);
146  }
147 }

149 void
150 EndDeviceLoraPhy::StartReceive (Ptr<Packet> packet, double rxPowerDbm,
151  uint8_t sf, Time duration, double frequencyMHz)
152 {
153 
154  NS_LOG_FUNCTION (this << packet << rxPowerDbm << unsigned (sf) << duration <<
155  frequencyMHz);
156 
157  // Notify the LoraInterferenceHelper of the impinging signal, and remember
158  // the event it creates. This will be used then to correctly handle the end
159  // of reception event.
160  //
161  // We need to do this regardless of our state or frequency, since these could
162  // change (and making the interference relevant) while the interference is
163  // still incoming.
164 
165  Ptr<LoraInterferenceHelper::Event> event;
166  event = m_interference.Add (duration, rxPowerDbm, sf, packet, frequencyMHz);
167 
168  // Switch on the current PHY state
169  switch (m_state)
170  {
171  // In the SLEEP, TX and RX cases we cannot receive the packet: we only add
172  // it to the list of interferers and do not schedule an EndReceive event for
173  // it.
174  case SLEEP:
175  {
176  NS_LOG_INFO ("Dropping packet because device is in SLEEP state");
177  break;
178  }
179  case TX:
180  {
181  NS_LOG_INFO ("Dropping packet because device is in TX state");
182  break;
183  }
184  case RX:
185  {
186  NS_LOG_INFO ("Dropping packet because device is already in RX state");
187  break;
188  }
189  // If we are in STANDBY mode, we can potentially lock on the currently
190  // incoming transmission
191  case STANDBY:
192  {
193  // There are a series of properties the packet needs to respect in order
194  // for us to be able to lock on it:
195  // - It's on frequency we are listening on
196  // - It uses the SF we are configured to look for
197  // - Its receive power is above the device sensitivity for that SF
198 
199  // Flag to signal whether we can receive the packet or not
200  bool canLockOnPacket = true;
201 
202  // Save needed sensitivity
203  double sensitivity = EndDeviceLoraPhy::sensitivity[unsigned(sf)-7];
204 
205  // Check frequency
207  if (!IsOnFrequency (frequencyMHz))
208  {
209  NS_LOG_INFO ("Packet lost because it's on frequency " <<
210  frequencyMHz << " MHz and we are listening at " <<
211  m_frequency << " MHz");
212 
213  // Fire the trace source for this event.
214  if (m_device)
215  {
216  m_wrongFrequency (packet, m_device->GetNode ()->GetId ());
217  }
218  else
219  {
220  m_wrongFrequency (packet, 0);
221  }
222 
223  canLockOnPacket = false;
224  }
225 
226  // Check Spreading Factor
228  if (sf != m_sf)
229  {
230  NS_LOG_INFO ("Packet lost because it's using SF" << unsigned(sf) <<
231  ", while we are listening for SF" << unsigned(m_sf));
232 
233  // Fire the trace source for this event.
234  if (m_device)
235  {
236  m_wrongSf (packet, m_device->GetNode ()->GetId ());
237  }
238  else
239  {
240  m_wrongSf (packet, 0);
241  }
242 
243  canLockOnPacket = false;
244  }
245 
246  // Check Sensitivity
248  if (rxPowerDbm < sensitivity)
249  {
250  NS_LOG_INFO ("Dropping packet reception of packet with sf = " <<
251  unsigned(sf) << " because under the sensitivity of " <<
252  sensitivity << " dBm");
253 
254  // Fire the trace source for this event.
255  if (m_device)
256  {
257  m_underSensitivity (packet, m_device->GetNode ()->GetId ());
258  }
259  else
260  {
261  m_underSensitivity (packet, 0);
262  }
263 
264  canLockOnPacket = false;
265  }
266 
267  // Check if one of the above failed
269  if (canLockOnPacket)
270  {
271  // Switch to RX state
272  // EndReceive will handle the switch back to STANDBY state
273  SwitchToRx ();
274 
275  // Schedule the end of the reception of the packet
276  NS_LOG_INFO ("Scheduling reception of a packet. End in " <<
277  duration.GetSeconds () << " seconds");
278 
279  Simulator::Schedule (duration, &LoraPhy::EndReceive, this, packet,
280  event);
281 
282  // Fire the beginning of reception trace source
283  m_phyRxBeginTrace (packet);
284  }
285  }
286  }

289 void
290 EndDeviceLoraPhy::EndReceive (Ptr<Packet> packet,
291  Ptr<LoraInterferenceHelper::Event> event)
292 {
293  NS_LOG_FUNCTION (this << packet << event);
294 
295  // Fire the trace source
296  m_phyRxEndTrace (packet);
297 
298  // Call the LoraInterferenceHelper to determine whether there was destructive
299  // interference on this event.
300  bool packetDestroyed = m_interference.IsDestroyedByInterference (event);
301 
302  // Fire the trace source if packet was destroyed
303  if (packetDestroyed)
304  {
305  NS_LOG_INFO ("Packet destroyed by interference");
306 
307  if (m_device)
308  {
309  m_interferedPacket (packet, m_device->GetNode ()->GetId ());
310  }
311  else
312  {
313  m_interferedPacket (packet, 0);
314  }
315 
316  }
317  else
318  {
319  NS_LOG_INFO ("Packet received correctly");
320 
321  if (m_device)
322  {
323  m_successfullyReceivedPacket (packet, m_device->GetNode ()->GetId ());
324  }
325  else
326  {
327  m_successfullyReceivedPacket (packet, 0);
328  }
329 
330  // If there is one, perform the callback to inform the upper layer
331  if (!m_rxOkCallback.IsNull ())
332  {
333  m_rxOkCallback (packet);
334  }
335 
336  }
337  // Automatically switch to Standby in either case
338  SwitchToStandby ();
339 }
