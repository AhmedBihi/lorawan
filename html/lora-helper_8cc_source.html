41  NS_LOG_FUNCTION_NOARGS ();

43  NetDeviceContainer devices;
44 
45  // Go over the various nodes in which to install the NetDevice
46  for (NodeContainer::Iterator i = c.Begin (); i != c.End (); ++i)
47  {
48  Ptr<Node> node = *i;
49 
50  // Create the LoraNetDevice
51  Ptr<LoraNetDevice> device = CreateObject<LoraNetDevice> ();
52 
53  // Create the PHY
54  Ptr<LoraPhy> phy = phyHelper.Create (node, device);
55  NS_ASSERT (phy != 0);
56  device->SetPhy (phy);
57  NS_LOG_DEBUG ("Done creating the PHY");
58 
59  // Create the MAC
60  Ptr<LoraMac> mac = macHelper.Create (node, device);
61  NS_ASSERT (mac != 0);
62  mac->SetPhy (phy);
63  NS_LOG_DEBUG ("Done creating the MAC");
64  device->SetMac (mac);
65 
66  node->AddDevice (device);
67  devices.Add (device);
68  NS_LOG_DEBUG ("node=" << node << ", mob=" << node->GetObject<MobilityModel> ());
69  }

71 }
