151 // LoRa Collision Matrix (Goursaud)
152 // Values are inverted w.r.t. the paper since here we interpret this as an
153 // _isolation_ matrix instead of a cochannel _rejection_ matrix like in
154 // Goursaud's paper.
155 const double LoraInterferenceHelper::collisionSnir[6][6] =
156 {
157 // SF7 SF8 SF9 SF10 SF11 SF12
158  { 6, -16, -18, -19, -19, -20}, // SF7

161  {-30, -30, -30, 6, -26, -28}, // SF10

169 LoraInterferenceHelper::Add (Time duration, double rxPower,
170  uint8_t spreadingFactor, Ptr<Packet> packet,
171  double frequencyMHz)
172 {
173 
174  NS_LOG_FUNCTION (this << duration.GetSeconds () << rxPower << unsigned
175  (spreadingFactor) << packet << frequencyMHz);
176 
177  // Create an event based on the parameters
178  Ptr<LoraInterferenceHelper::Event> event =
179  Create<LoraInterferenceHelper::Event> (duration, rxPower, spreadingFactor,
180  packet, frequencyMHz);
181 
182  // Add the event to the list
183  m_events.push_back (event);
184 
185  // Clean the event list
186  if (m_events.size () > 100)
187  {
188  CleanOldEvents ();
189  }
190 
191  return event;
192 }

199  // Cycle the events, and clean up if an event is old.
200  for (auto it = m_events.begin (); it != m_events.end ();)

206  it++;

230 uint8_t
231 LoraInterferenceHelper::IsDestroyedByInterference
232  (Ptr<LoraInterferenceHelper::Event> event)
233 {
234  NS_LOG_FUNCTION (this << event);
235 
236  NS_LOG_INFO ("Current number of events in LoraInterferenceHelper: " << m_events.size ());
237 
238  // We want to see the interference affecting this event: cycle through events
239  // that overlap with this one and see whether it survives the interference or
240  // not.
241 
242  // Gather information about the event
243  double rxPowerDbm = event->GetRxPowerdBm ();
244  uint8_t sf = event->GetSpreadingFactor ();
245  double frequency = event->GetFrequency ();
246 
247  // Handy information about the time frame when the packet was received
248  Time now = Simulator::Now ();
249  Time duration = event->GetDuration ();
250  Time packetStartTime = now - duration;
251  Time packetEndTime = now;
252 
253  // Get the list of interfering events
254  std::list<Ptr<LoraInterferenceHelper::Event> >::iterator it;
255 
256  // Energy for interferers of various SFs
257  std::vector<double> cumulativeInterferenceEnergy (6,0);
258 
259  // Cycle over the events
260  for (it = m_events.begin (); it != m_events.end ();)
261  {
262  // Pointer to the current interferer
263  Ptr< LoraInterferenceHelper::Event > interferer = *it;
264 
265  // Only consider the current event if the channel is the same: we
266  // assume there's no interchannel interference. Also skip the current
267  // event if it's the same that we want to analyze.
268  if (!(interferer->GetFrequency () == frequency) || interferer == event)
269  {
270  NS_LOG_DEBUG ("Different channel");
271  it++;
272  continue; // Continues from the first line inside the for cycle
273  }
274 
275  NS_LOG_DEBUG ("Interferer on same channel");
276 
277  // Gather information about this interferer
278  uint8_t interfererSf = interferer->GetSpreadingFactor ();
279  double interfererPower = interferer->GetRxPowerdBm ();
280  Time interfererStartTime = interferer->GetStartTime ();
281  Time interfererEndTime = interferer->GetEndTime ();
282 
283  NS_LOG_INFO ("Found an interferer: sf = " << unsigned(interfererSf)
284  << ", power = " << interfererPower
285  << ", start time = " << interfererStartTime
286  << ", end time = " << interfererEndTime);
287 
288  // Compute the fraction of time the two events are overlapping
289  Time overlap = GetOverlapTime (event, interferer);
290 
291  NS_LOG_DEBUG ("The two events overlap for " << overlap.GetSeconds () << " s.");
292 
293  // Compute the equivalent energy of the interference
294  // Power [mW] = 10^(Power[dBm]/10)
295  // Power [W] = Power [mW] / 1000
296  double interfererPowerW = pow (10, interfererPower/10) / 1000;
297  // Energy [J] = Time [s] * Power [W]
298  double interferenceEnergy = overlap.GetSeconds () * interfererPowerW;
299  cumulativeInterferenceEnergy.at (unsigned(interfererSf)-7) += interferenceEnergy;
300  NS_LOG_DEBUG ("Interferer power in W: " << interfererPowerW);
301  NS_LOG_DEBUG ("Interference energy: " << interferenceEnergy);
302  it++;
303  }
304 
305  // For each SF, check if there was destructive interference
306  for (uint8_t currentSf = uint8_t (7); currentSf <= uint8_t (12); currentSf++)
307  {
308  NS_LOG_DEBUG ("Cumulative Interference Energy: " <<
309  cumulativeInterferenceEnergy.at (unsigned(currentSf)-7));
310 
311  // Use the computed cumulativeInterferenceEnergy to determine whether the
312  // interference with this SF destroys the packet
313  double signalPowerW = pow (10, rxPowerDbm/10) / 1000;
314  double signalEnergy = duration.GetSeconds () * signalPowerW;
315  NS_LOG_DEBUG ("Signal power in W: " << signalPowerW);
316  NS_LOG_DEBUG ("Signal energy: " << signalEnergy);
317 
318  // Check whether the packet survives the interference of this SF
319  double snirIsolation = collisionSnir [unsigned(sf)-7][unsigned(currentSf)-7];
320  NS_LOG_DEBUG ("The needed isolation to survive is "
321  << snirIsolation << " dB");
322  double snir = 10*log10 (signalEnergy/cumulativeInterferenceEnergy.at (unsigned(currentSf)-7));
323  NS_LOG_DEBUG ("The current SNIR is " << snir << " dB");
324 
325  if (snir >= snirIsolation)
326  {
327  // Move on and check the rest of the interferers
328  NS_LOG_DEBUG ("Packet survived interference");
329  }
330  else
331  {
332  NS_LOG_DEBUG ("Packet destroyed by interference with SF" <<
333  unsigned(currentSf));
334 
335  return currentSf;
336  }
337  }
338  // If we get to here, it means that the packet survived all interference
339  NS_LOG_DEBUG ("Packet survived all interference");
340 
341  // Since the packet was not destroyed, we return 0.
342  return uint8_t (0);
343 }

353 Time
354 LoraInterferenceHelper::GetOverlapTime (Ptr<LoraInterferenceHelper::Event> event1,
355  Ptr<LoraInterferenceHelper::Event> event2)
356 {
357  NS_LOG_FUNCTION_NOARGS ();
358 
359  // Create the value we will return later
360  Time overlap;
361 
362  // Get handy values
363  Time s1 = event1->GetStartTime (); // Start times
364  Time s2 = event2->GetStartTime ();
365  Time e1 = event1->GetEndTime (); // End times
366  Time e2 = event2->GetEndTime ();
367 
368  // Event1 starts before Event2
369  if (s1 < s2)
370  {
371  // Non-overlapping events
372  if (e1 < s2)
373  {
374  overlap = Seconds (0);
375  }
376  // event1 contains event2
377  else if (e1 >= e2)
378  {
379  overlap = e2 - s2;
380  }
381  // Partially overlapping events
382  else
383  {
384  overlap = e1 - s2;
385  }
386  }
387  // Event2 starts before Event1
388  else
389  {
390  // Non-overlapping events
391  if (e2 < s1)
392  {
393  overlap = Seconds (0);
394  }
395  // event2 contains event1
396  else if (e2 >= e1)
397  {
398  overlap = e1 - s1;
399  }
400  // Partially overlapping events
401  else
402  {
403  overlap = e2 - s1;
404  }
405  }
406  return overlap;
407 }
