77 Ptr<Application>
78 PeriodicSenderHelper::InstallPriv (Ptr<Node> node) const
79 {
80  NS_LOG_FUNCTION (this << node);
81 
82  Ptr<PeriodicSender> app = m_factory.Create<PeriodicSender> ();
83 
84  Time interval;
85  if (m_period == Seconds (0))
86  {
87  double intervalProb = m_intervalProb->GetValue ();
88  NS_LOG_DEBUG ("IntervalProb = " << intervalProb);
89 
90  // Based on TR 45.820
91  if (intervalProb < 0.4)
92  {
93  interval = Days (1);
94  }
95  else if (0.4 <= intervalProb && intervalProb < 0.8)
96  {
97  interval = Hours (2);
98  }
99  else if (0.8 <= intervalProb && intervalProb < 0.95)
100  {
101  interval = Hours (1);
102  }
103  else interval = Minutes (30);
104  }
105  else interval = m_period;
106 
107  app->SetInterval (interval);
108  NS_LOG_DEBUG ("Created an application with interval = " <<
109  interval.GetHours () << " hours");
110 
111  app->SetInitialDelay (Seconds (m_initialDelay->GetValue (0, interval.GetSeconds ())));
112 
113  app->SetNode (node);
114  node->AddApplication (app);
115 
116  return app;
117 }
