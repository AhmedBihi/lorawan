81 NetworkServerHelper::InstallPriv (Ptr<Node> node)
82 {
83  NS_LOG_FUNCTION (this << node);
84 
85  Ptr<SimpleNetworkServer> app = m_factory.Create<SimpleNetworkServer> ();
86 
87  app->SetNode (node);
88  node->AddApplication (app);
89 
90  // Cycle on each gateway
91  for (NodeContainer::Iterator i = m_gateways.Begin ();
92  i != m_gateways.End ();
93  i++)
94  {
95  // Add the connections with the gateway
96  // Create a PointToPoint link between gateway and NS
97  NetDeviceContainer container = p2pHelper.Install (node, *i);
98 
99  // Add the gateway to the NS list
100  app->AddGateway (*i, container.Get (0));
101  }
102 
103  // Link the SimpleNetworkServer to its NetDevices
104  for (uint32_t i = 0; i < node->GetNDevices (); i++)
105  {
106  Ptr<NetDevice> currentNetDevice = node->GetDevice (i);
107  currentNetDevice->SetReceiveCallback (MakeCallback
108  (&SimpleNetworkServer::Receive,
109  app));
110  }
111 
112  // Add the end devices
113  app->AddNodes (m_endDevices);
114 
115  return app;
116 }
