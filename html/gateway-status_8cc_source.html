82 bool
83 GatewayStatus::IsAvailableForTransmission (double frequency)
84 {
85  // We can't send multiple packets at once, see SX1301 V2.01 page 29
86 
87  // Check that the gateway was not already "booked"
88  if (m_nextTransmissionTime > Simulator::Now () - MilliSeconds (1))
89  {
90  NS_LOG_INFO ("This gateway is already booked for a transmission");
91  return false;
92  }
93 
94  // Check that the gateway is not already in TX mode
95  if (m_gatewayMac->IsTransmitting ())
96  {
97  NS_LOG_INFO ("This gateway is currently transmitting");
98  return false;
99  }
100 
101  // Check that the gateway is not constrained by the duty cycle
102  Time waitingTime = m_gatewayMac->GetWaitingTime (frequency);
103  if (waitingTime > Seconds (0))
104  {
105  NS_LOG_INFO ("Gateway cannot be used because of duty cycle");
106  NS_LOG_INFO ("Waiting time at current GW: " << waitingTime.GetSeconds ()
107  << " seconds");
108 
109  return false;
110  }
111 
112  return true;
113 }
